def corner_to_centroid(z, n_x, n_y):

    '''
    A function to process surface file modeled at grid corners and interpolate the centroids.
    The input z is the modeled surface elevation/depth with size of (n_x+1)*(n_y+1). The output
    well be a n_x*n_y list
    
    z : z picks for the modeled surface. Tidy (long-form) 1D data where a single column of the variable exists with each row is
            an observation. A pandas dataframe/series or numpy array can be passed.
    n_x (int): number of grid cells
    n_y (int): number of grid cells

    Example: 
        Plotting a collection of horizontal well trajectories:

        >>> X = PythonTools.corner_to_centroid(surface_df['Top_Surface'], n_x=50, n_y=60)


    '''
    import numpy as np

    if  len(z)==(n_x+1)*(n_y+1):
        pass
    else:
        raise ValueError('The provided surface is not modeled at corners based on provided n_x and n_y')


    list_centroids = []

    for j in range (1, n_y+1):

        for i in range (1, n_x+1):

            c_1 = i + (j-1)*(n_x+1) # index for corner 1

            c_2 = (i+1) + (j-1)*(n_x+1) # index for corner 2

            c_3 = i + (j)*(n_x+1) # index for corner 3

            c_4 = (i+1) + (j)*(n_x+1) # index for corner 4

            list_centroids.append(np.mean([z[c_1-1], z[c_2-1], z[c_3-1], z[c_4-1]]))

    return list_centroids


def surface_to_vtk( z, griddef,  flname='Surface.vtk', fmt='%.5f', centroid_always = True):

    '''
    A function to write a vtk file that can be used to visualize a surface based on triangulation.
    
    z (long-form): The elevation/depth picks for the surface. 1D data where a single column of the variable exists with each row is an observation. A pandas dataframe/series or numpy array can be passed. 
    griddef(pygeostat griddef): A pygeostat object that contains grid specifications.
    flname (string): Name of the output file
    fmt (string): Format to write the vtk file
    centroid_always(bool): an option to determine if the surface was modeled at centroids (i.e. centroid_always = True) or grid corners (i.e. centroid_always = False)

    Example: 
        Plotting a collection of horizontal well trajectories:
        >>> import PythonTools as PT
        >>> grid_model = PT.GridDef([nx, xmn, xsiz, ny, ymn, ysiz,nz, zmn, zsiz])
        >>> PT.surface_to_vtk(z, grid_model, flname='Surface.vtk')

        .. image:: ./figures/Surface.png
        .. image:: ./figures/Surface2.png

    '''

    from scipy.spatial import Delaunay
    import numpy as np
    import pandas as pd
    

    # set the griddef to be a surface one (i.e. 2D)
    try:
        n_z = griddef.nz
        griddef.nz = 1
        n_x = griddef.nx
        n_y = griddef.ny
        x_mn = griddef.xmn
        y_mn = griddef.ymn
        x_siz = griddef.xsiz
        y_siz = griddef.ysiz

    except:
        raise TypeError('gridded must be a pygeostat griddef')

    try:
        z = np.array(z)
    except:
        raise TypeError('The surface must be a pandas DataFrame or numpy array')


    if len(z) == (n_x+1)*(n_y+1): # Surface is modeled at the corner of grid cells

        print('surface is modeled at grid corners')
        
        if not centroid_always:

            print('Interpolation is required to calculate centroids based on corners')

            # interpolate to find the centroids
            List = corner_to_centroid(z, n_x, n_y)
            z = np.array(List)
        else:
            griddef.nx = n_x + 1
            griddef.ny = n_y + 1
            griddef.xmn = x_mn - (x_siz/2)
            griddef.ymn = y_mn - (y_siz/2)


    elif len(z) == n_x*n_y:

        pass

    else:
        raise ValueError('The provided surface does not match the griddef')


    points = griddef.gengridpoints()

    # restore griddef
    #--------------------
    griddef.nz = n_z
    griddef.nx = n_x
    griddef.ny = n_y
    griddef.xmn = x_mn
    griddef.ymn = y_mn
    
    x = points[:,0]
    y=points[:,1]
    # Data sanity check
    try:
        x = x.ravel()
        y = y.ravel()
        z = z.ravel()
    except Exception as e:
        raise ValueError('Wrong data type s provided for X or Y or Z')

    Input=np.zeros((len(x),2))
    Input[:,0]=x;
    Input[:,1]=y;
    TRIANG = Delaunay(Input)
    TRIANG=TRIANG.simplices.copy()
    
    nbpoint=len(x);
    
    if np.mod(nbpoint,3)==1:
        x=np.append(x,[0,0]);
        y=np.append(y,[0,0]);
        z=np.append(z,[0,0]);
    elif np.mod(nbpoint,3)==2:
        x=np.append(x,[0]);
        y=np.append(y,[0]);
        z=np.append(z,[0]);
        
    nbpoint=len(x);
    nTRIANG=len(TRIANG);
    with open(flname, 'w') as outfl:
        outfl.write('# vtk DataFile Version 3.0\n'
                    'vtk output\n'
                    'ASCII\n'
                    'DATASET POLYDATA\n');
        sTRIANGng='POINTS %d float\n'%nbpoint
        outfl.write(sTRIANGng);
        DF=pd.DataFrame()
        DF['X1']=x[:-2:3]
        DF['Y1']=y[:-2:3]
        DF['Z1']=z[:-2:3]
        
        DF['X2']=x[1:-1:3]
        DF['Y2']=y[1:-1:3]
        DF['Z2']=z[1:-1:3]
        
        DF['X3']=x[2::3]
        DF['Y3']=y[2::3]
        DF['Z3']=z[2::3]
        DF.to_csv(outfl, header=False, index=False, sep=' ',
                                           float_format=fmt)
        
        sTRIANGng='POLYGONS %d %d\n'%(nTRIANG,4*nTRIANG)
        outfl.write(sTRIANGng);
        A3=3*np.ones((len(TRIANG),1))
        TRIANG=np.append(A3,TRIANG,axis=1)
        DF=pd.DataFrame(data=TRIANG)
        DF.to_csv(outfl, header=False, index=False, sep=' ',
                                           float_format='%d')
