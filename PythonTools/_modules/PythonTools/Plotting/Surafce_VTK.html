from __future__ import absolute_import, division, print_function
__author__ = 'Mostafa Hadavand'
__date__ = '2017-03-03'
__version__ = '1.00'



def Surafce_VTK(x, y, z, flname='Surface.vtk', fmt='%.5f'):

    '''
    A function to write a vtk file that can be used to visualize a surface based on triangulation.
    
    data (pandas.DataFrame): Pandas dataframe containing array of data values
    x (string): X coordinate
    y (string): Y coordinate
    z (string): Z picks
    flname (string): Name of the output file
    fmt (string): Format to write the vtk file

    Example: 
        Plotting a collection of horizontal well trajectories:

        >>> X = np.arange(-5, 5, 0.25)
        >>> Y = np.arange(-5, 5, 0.25)
        >>> X, Y = np.meshgrid(X, Y)
        >>> R = np.sqrt(X**2 + Y**2)
        >>> Z = np.sin(R)
        >>> PythonTools.Surafce_VTK(x=X, y=Y, z=Z)

        .. image:: ./_figures/Surface.png
    '''

    from scipy.spatial import Delaunay
    import numpy as np
    import pandas as pd
    

    # Data sanity check
    try:
        x = x.ravel()
        y = y.ravel()
        z = z.ravel()
    except Exception as e:
        raise ValueError('Wrong data type s provided for X or Y or Z')

    Input=np.zeros((len(x),2))
    Input[:,0]=x;
    Input[:,1]=y;
    TRIANG = Delaunay(Input)
    TRIANG=TRIANG.simplices.copy()
    
    nbpoint=len(x);
    
    if np.mod(nbpoint,3)==1:
        x=np.append(x,[0,0]);
        y=np.append(y,[0,0]);
        z=np.append(z,[0,0]);
    elif np.mod(nbpoint,3)==2:
        x=np.append(x,[0]);
        y=np.append(y,[0]);
        z=np.append(z,[0]);
        
    nbpoint=len(x);
    nTRIANG=len(TRIANG);
    with open(flname, 'w') as outfl:
        outfl.write('# vtk DataFile Version 3.0\n'
                    'vtk output\n'
                    'ASCII\n'
                    'DATASET POLYDATA\n');
        sTRIANGng='POINTS %d float\n'%nbpoint
        outfl.write(sTRIANGng);
        DF=pd.DataFrame()
        DF['X1']=x[:-2:3]
        DF['Y1']=y[:-2:3]
        DF['Z1']=z[:-2:3]
        
        DF['X2']=x[1:-1:3]
        DF['Y2']=y[1:-1:3]
        DF['Z2']=z[1:-1:3]
        
        DF['X3']=x[2::3]
        DF['Y3']=y[2::3]
        DF['Z3']=z[2::3]
        DF.to_csv(outfl, header=False, index=False, sep=' ',
                                           float_format=fmt)
        
        sTRIANGng='POLYGONS %d %d\n'%(nTRIANG,4*nTRIANG)
        outfl.write(sTRIANGng);
        A3=3*np.ones((len(TRIANG),1))
        TRIANG=np.append(A3,TRIANG,axis=1)
        DF=pd.DataFrame(data=TRIANG)
        DF.to_csv(outfl, header=False, index=False, sep=' ',
                                           float_format='%d')
