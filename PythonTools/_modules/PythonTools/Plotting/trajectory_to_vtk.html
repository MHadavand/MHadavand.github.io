def ismember(A, B, Null=-999):
    flag=[]
    member=[]
    for i in A:
        if (i in B):
            flag.append(1)
            member.append(i)
        else:
            flag.append(0)
            member.append(Nulle)
    return flag,member


def trajectory_to_vtk(data = None, x = 'X', y = 'Y', z = 'Z', var = 'INCL', well_id = 'ID', vtp_folder = 'Wells_VTK',
             categorical = False, output_file = './Wells.vtm'):
    
    '''
    A function to write a vtk file that can be used to visualize well trajectory
    
    data (pandas.DataFrame): Pandas dataframe containing array of data values
    x (string): Name of X coordinate column
    y (string): Name of Y coordinate column
    z (string): Name of Z coordinate column
    var (string): Name of variable column
    well_id (string): Name of well id column

    Example: 
        Plotting a collection of horizontal well trajectories:
        
        >>> PythonTools.trajectory_to_vtk(data=DF, x = 'X', y = 'Y', z = 'Z', var = 'INCL', well_id = 'ID')

        .. image:: ./figures/WellVtk.png
    '''

    if (data is not None) and (not isinstance(data, pd.DataFrame)):
        try:
            DF = pd.DataFrame(data=data)
        except Exception as ex:
            print('Exception: %s'%type(ex))
            print('Exception description: %s'%ex)
            raise ValueError('Please provide a pandas dataframe for input data')
    else:
        DF = data
            

    if categorical:
        rocktypes = np.unique(DF[var])
        

    if not os.path.exists(os.path.dirname(output_file)):
        os.makedirs(os.path.dirname(output_file))

    dh, dhk = np.unique(DF[well_id], return_index=True) # Get the head/tail coordinate based 
    ndh = len(dh)
    
    if categorical:
    
        with open(os.path.join('\\',os.getcwd(),output_file),'w') as mainfile:
            
            # Writing the header file with list of all wells
            mainfile.write('<VTKFile type="vtkMultiBlockDataSet" version="1.0" byte_order="LittleEndian" header_type="UInt64">\n')
            mainfile.write('<vtkMultiBlockDataSet>\n')
            Well_Directory=os.path.join(os.path.dirname(output_file),vtp_folder)
            if not os.path.exists(Well_Directory):
                os.makedirs(Well_Directory)
            for i in range(ndh):
                dat = pd.DataFrame(columns = ['X', 'Y', 'Z'])
                dat['X'] = DF[x][DF[well_id]==int(dh[i])]
                dat['Y'] = DF[y][DF[well_id]==int(dh[i])]
                dat['Z'] = DF[z][DF[well_id]==int(dh[i])]
                dat.reset_index(drop=True, inplace=True)
                Np = len(dat['X'])
                I = np.append([0],np.reshape(np.matlib.repmat(np.arange(1,Np-1), 2, 1),(Np-2)*2,1)) 
                I = np.append(I,[Np-1])
                J = np.reshape(np.matlib.repmat(np.arange(1,Np), 2, 1),(Np-1)*2,1) - 1
                mainfile.write('<DataSet index="' + str(i) + '" file="' + vtp_folder + '/dh_' + str(i+1) + '.vtp">\n');
                mainfile.write('</DataSet>\n')
                
                with open(os.path.join('\\',os.getcwd(),Well_Directory+'/dh_%i.vtp'%(i+1)),'w') as WellFile:
                    
                    WellFile.write('<VTKFile type="PolyData" version="1.0" byte_order="LittleEndian" header_type="UInt64">\n')
                    WellFile.write('<PolyData>\n')
                    WellFile.write('<Piece NumberOfPoints="' + str((Np-1)*2) + '" NumberOfVerts="0" NumberOfLines="' + str(Np-1) + '" NumberOfStrips="0" NumberOfPolys="0">\n')
                    WellFile.write('<Points>\n')
                    WellFile.write('  <DataArray type="Float32" Name="Points" NumberOfComponents="3" format="ascii">\n')
                    dat.ix[I].to_csv(WellFile, header=False, index=False, sep='\t', float_format='%13.6f')
                    WellFile.write('</DataArray>\n');
                    WellFile.write('</Points>\n');
                    WellFile.write('<PointData Scalars="DATA">\n');
                    WellFile.write('  <DataArray type="Int32" Name="Categorical Propety" format="ascii">\n');
                    
                    temp = DF[var][DF[well_id]==dh[i]]
                    temp.reset_index(drop=True, inplace=True)
                    flag, member = ismember(temp,rocktypes)
                    MembDF=pd.DataFrame(data=member,columns=['member'])
                    MembDF.ix[J].to_csv(WellFile, header=False, index=False, sep='\t', float_format='%i')
                    
                    WellFile.write('  </DataArray>\n');
                    WellFile.write('</PointData>\n');

                    WellFile.write(f'<Lines>\n');
                    WellFile.write('  <DataArray type="Int64" Name="connectivity" format="ascii">\n');
                    WellFile.write('    ' + str(np.arange(0,len(I))).replace(']','').replace('[','') + '\n')
                    WellFile.write('  </DataArray>\n');
                    WellFile.write('  <DataArray type="Int64" Name="offsets" format="ascii">\n');
                    WellFile.write('    ' + str(np.arange(2,2*(Np),2)).replace(']','').replace('[','') + '\n');
                    WellFile.write('  </DataArray>\n');
                    WellFile.write('</Lines>\n');
                    WellFile.write('</Piece>\n');
                    WellFile.write('</PolyData>\n');
                    WellFile.write('</VTKFile>\n');

            mainfile.write('</vtkMultiBlockDataSet>\n');
            mainfile.write('</VTKFile>\n');
    
    else:
        
        with open(os.path.join('\\',os.getcwd(),output_file),'w') as mainfile:
            
            # Writing the header file with list of all wells
            mainfile.write('<VTKFile type="vtkMultiBlockDataSet" version="1.0" byte_order="LittleEndian" header_type="UInt64">\n')
            mainfile.write('<vtkMultiBlockDataSet>\n')
            Well_Directory=os.path.join(os.path.dirname(output_file),vtp_folder)
            if not os.path.exists(Well_Directory):
                os.makedirs(Well_Directory)
            for i in range(ndh):
                dat = pd.DataFrame(columns = ['X', 'Y', 'Z'])
                dat['X'] = DF[x][DF[well_id]==int(dh[i])]
                dat['Y'] = DF[y][DF[well_id]==int(dh[i])]
                dat['Z'] = DF[z][DF[well_id]==int(dh[i])]
                dat.reset_index(drop=True, inplace=True)
                Np = len(dat['X'])
                I = np.arange(1,Np+1)
                J = np.arange(1,Np+1)
                targetfile = os.path.join('\\',os.getcwd(),Well_Directory+'/dh_%i.vtp'%(i+1))
                mainfile.write('<DataSet index="' + str(i) + '" file="' + vtp_folder + '/dh_' + str(i+1) + '.vtp">\n');
                mainfile.write('</DataSet>\n')
                with open(targetfile,'w') as WellFile:
                    WellFile.write('<VTKFile type="PolyData" version="1.0" byte_order="LittleEndian" header_type="UInt64">\n')
                    WellFile.write('<PolyData>\n')
                    WellFile.write('<Piece NumberOfPoints="' + str(Np) + '" NumberOfVerts="0" NumberOfLines="' + str(1) + '" NumberOfStrips="0" NumberOfPolys="0">\n')
                    WellFile.write('<Points>\n')
                    WellFile.write('  <DataArray type="Float32" Name="Points" NumberOfComponents="3" format="ascii">\n')
                    dat.to_csv(WellFile, header=False, index=False, sep='\t', float_format='%13.6f')
                    WellFile.write('</DataArray>\n');
                    WellFile.write('</Points>\n');
                    WellFile.write('<PointData Scalars="DATA">\n');
                    WellFile.write('  <DataArray type="Float32" Name="Continuos Propety" format="ascii">\n');
                    temp = DF[var][DF[well_id]==dh[i]]
                    temp.reset_index(drop=True, inplace=True)
                    DF[var][DF[well_id]==dh[i]].to_csv(WellFile, header=False, index=False, sep='\t', float_format='%f')
                    
                    WellFile.write('  </DataArray>\n');
                    WellFile.write('</PointData>\n');

                    WellFile.write(f'<Lines>\n');
                    WellFile.write('  <DataArray type="Int64" Name="connectivity" format="ascii">\n');
                    WellFile.write('    ' + str(np.arange(0,len(I))).replace(']','').replace('[','') + '\n')
                    WellFile.write('  </DataArray>\n');
                    WellFile.write('  <DataArray type="Int64" Name="offsets" format="ascii">\n');
                    WellFile.write('    ' + str(Np) + '\n');
                    WellFile.write('  </DataArray>\n');
                    WellFile.write('</Lines>\n');
                    WellFile.write('</Piece>\n');
                    WellFile.write('</PolyData>\n');
                    WellFile.write('</VTKFile>\n');
            mainfile.write('</vtkMultiBlockDataSet>\n');
            mainfile.write('</VTKFile>\n')
