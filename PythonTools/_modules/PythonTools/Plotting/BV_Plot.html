"""BV_Plot"""
from __future__ import absolute_import, division, print_function
__author__ = 'Mostafa Hadavand'
__date__ = '2016-15-02'
__version__ = '1.200'

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.cm as cm
import matplotlib as mpl
from mpl_toolkits.axes_grid1 import make_axes_locatable
from . import kde
import os

def bv_plot(x, y, xlabel='Variable 1', ylabel='Variable 2', bw=0.2, figsize=(10,10), cmap='hot_r', contour=False, ax=None,
           s_size=100, fontsize=11, label_font_size= 11, alpha=1.0, kde_label=True, cbar=True, stat_xy=(1.05,0.99)):

    """
    Bivariate probability plot based on kernel density estimate.  If the user sets a negative value for
       bandwidth, it will be determined automatically. The estimation works best for a unimodal distribution; bimodal or 
       multi-modal distributions tend to be oversmoothed..

    Parameters:
        x: Tidy (long-form): Variable 1. 1D data where a single column of the variable exists with each row is
            an observation. A pandas dataframe/series or numpy array can be passed.
        y: Tidy (long-form): Variable 2. 1D data where a single column of the variable exists with each row is
            an observation. A pandas dataframe/series or numpy array can be passed.
        xlabel (string): Lable for the x axis
        ylabel (string): Lable for the y axis
        bw (float): Bandwidth for kernel density calculation (negative value for automated pick).
        cmap ('str'): Matplotlib color map

    .. codeauthor:: Mostafa Hadavand 2015-18-8
    
    Example: 
        Plotting a bivariate normal distribution:
        
        >>> PythonTools.bv_plot (x, y, bw = -1)

        .. image:: ./figures/BV_Plot.png

    """
    
    
    Rho = np.corrcoef(x, y)
    txtstats = r'$\rho$ = {CC}'.format(CC=str("%.2f" % Rho[0,1]))

    if ax is None:
        fig, ax = plt.subplots(1, figsize=figsize)
        
    ax.spines['right'].set_visible(False)
    ax.spines['top'].set_visible(False)
    ax.xaxis.set_ticks_position('bottom')
    ax.yaxis.set_ticks_position('left')
    xmin=x.min()
    xmax=x.max()
    ymin=y.min()
    ymax=y.max()
    
    if not contour:
        values = np.vstack([x, y])
        positions = values
        kernel = kde.kde_scipy(values, positions, bandwidth=bw)
        cax=ax.scatter(x,y,c=kernel,cmap=cmap,s=s_size,marker='o',edgecolor='k')
        ax.set_xlim([xmin - np.sign(xmin)*0.05*xmin, xmax + np.sign(xmax)*0.05*xmax])
        ax.set_ylim([ymin - np.sign(ymin)*0.05*ymin, ymax + np.sign(ymax)*0.05*ymax])
        ax.set_xlabel(xlabel, fontsize=label_font_size)
        ax.set_ylabel(ylabel,fontsize=label_font_size)
        if cbar:
            cbar = plt.colorbar(cax, ticks=[np.amin(kernel), (np.amax(kernel)-np.amin(kernel))/2 ,np.amax(kernel)])
            if kde_label:
                cbar.ax.set_ylabel('KDE')
    else:
        A, B = np.mgrid[xmin:xmax:100j, ymin:ymax:100j]
        Positions = np.vstack([A.ravel(), B.ravel()])
        values = np.vstack([x, y])
        kernel = kde.kde_scipy(values, Positions, bandwidth=bw)
        Z = np.reshape(kernel.T, A.shape)
        image=ax.imshow(np.rot90(Z), cmap=cmap,interpolation='hermite',
               extent=[xmin, xmax, ymin, ymax],aspect='auto',alpha=alpha)
        cset = ax.contour(A, B, Z, zdir='z', offset=0, cmap=cmap,linewidths=3)
        ax.set_xlabel(xlabel)
        ax.set_ylabel(ylabel)
        ax.set_xlim([xmin, xmax])
        ax.set_ylim([ymin, ymax])
        divider = make_axes_locatable(ax)
        if cbar:
            cax = divider.append_axes("right", size=0.1,pad=0.05)
            cbar = plt.colorbar(image, ticks=[np.amin(Z), (np.amax(Z)-np.amin(Z))/2 ,np.amax(Z)],cax=cax)
            if kde_label:
                cbar.ax.set_ylabel('KDE')
    stat_xy=stat_xy
    ax.text(stat_xy[0], stat_xy[1], txtstats, va='top', ha='right', transform=ax.transAxes,
                            fontsize=fontsize)
    return 

Main_funcs = [bv_plot]
