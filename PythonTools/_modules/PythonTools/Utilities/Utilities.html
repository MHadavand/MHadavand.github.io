def mkdir(directory):

	"""
	This function check if the provided directory exists and if not it creates a new one
	Parameters:
		directory(str): directory to be checked and created.
	Example:

		>>> PythonTools.mkdir('E:/Git_Repo')
	"""

	import os

	if directory is None:
		raise ValueError ('Please provide the target directory!')

	if not isinstance(directory,str):
		raise TypeError ('The directory must be a string entry!')

	if not os.path.exists(directory):
		os.makedirs(directory)

def random_rgb_color(ncolors = 20, seed = 69069):

	"""
	This function generates an array of colors using numpy random number generator. The seed number can be used to control the random number generator sequence.

	Parameters:
		ncolors(int): Desired number of colors
		seed(int): Random number generator seed number
	Example:

		>>> RandomColor = PythonTools.random_rgb_color(ncolors = 20, seed = 69069)
	"""
	import numpy as np

	np.random.seed(seed)
	ListColor = []

	for i in range(ncolors):
		ListColor.append([np.random.randint(0, 255, 1)[0]/255, np.random.randint(0, 255, 1)[0]/255, np.random.randint(0, 255, 1)[0]/255])

	return ListColor



def save_plot(outfl=None, dpi=300, pad_inches =0.02 , **kwargs):
	
	"""
	This function can be used to save plots in different formats (i.e., png, jpg, pdf,...)

	Parameters:
		outfl(str): Output address (directory + file name)
		dpi(int): dpi for the saved plot
		pad_inches (float): Pad size around the plot
		**kwargs: Optional permissible keyword
	Example:

		>>> RandomColor = PythonTools.save_plot(outfl='E:/Git_Repo/Figure1.pdf', dpi=300, pad_inches =0.02)
	"""

	import os
	import matplotlib as mpl
	import matplotlib.pyplot as plt
	from matplotlib.backends.backend_pdf import PdfPages



	_, extension = os.path.splitext(outfl)

	if extension.upper() == '.PDF':
		
		pdfout = PdfPages (outfl)
		plt.savefig(pdfout, format='pdf', bbox_inches='tight', pad_inches=pad_inches, dpi=dpi, **kwargs)
		pdfout.close()
	else:
		plt.savefig(outfl, bbox_inches='tight', pad_inches=pad_inches, dpi=dpi, **kwargs)


def unzip_file(path_to_zip_file = None , destination = None):

	"""
	This function can be used to unzip a zip file

	Parameters:
		path_to_zip_file(str): Address of the zip file 
		destination(str): destination for unzipped file

	Example:

		>>> PythonTools.unzip_file(path_to_zip_file = 'Well_Reports.zip', destination = Well_Reports )
	"""

	if path_to_zip_file is None:
		raise ValueError('path_to_zip_file is required!')

	if destination is None:
		raise ValueError('destination is required!')

	import zipfile
	with zipfile.ZipFile(path_to_zip_file,'r') as MainFile:
		MainFile.extractall(destination)


def get_file_size(file_path = None, KB = 1024, scale = None, size_in_bytes_only = False):

	"""
	This function can be used to get a file path an return the file size

	Parameters:
		file_path(str): Path/address for the target file 
		KB(int): Measure of every kilo bytes (1024/1000)
		scale(str): Desired scale ('bytes', 'KB', 'MB', 'GB', 'TB'). 'TB' is used of scale is None

	Example:

		>>> PythonTools.get_file_size(file_path = 'Well_Reports.mdb',KB=1000, scale='MB')
	"""
	
	import os

	if file_path is None:
		raise ValueError('file_path is required!')

	try:
		file_path = str(file_path)
	except Exception as e:
		raise ValueError('file_path needs to be a string' + str(e))

	statinfo = os.stat(file_path)
	SizeinBytes = statinfo.st_size
	
	if size_in_bytes_only:
		return SizeinBytes
	
	for item in ['bytes', 'KB', 'MB', 'GB', 'TB']:
		if SizeinBytes < KB or item == scale:
			return "File size: %3.5f %s" % (SizeinBytes, item)
		SizeinBytes /= KB

def add_month(input_date = None, num_months = 1):

	"""
	This function can be used to increment datatime.date by specified number of months

	Parameters:
		input_date(datatime.date): input date 
		num_months(int): number of months for each increment

	Example:

		>>> PythonTools.add_month(input_date = datetime.date.today(), num_months = 1))
	"""

	import datetime
	import calendar

	if input_date is None:
		raise ValueError('input_date is required!')

	try:
		assert (isinstance(input_date, datetime.date))
	except:
		raise ValueError('input_date should be a datetime.date!')


	month = input_date.month - 1 + num_months
	year = input_date.year + month // 12
	month = month % 12 + 1
	day = min(input_date.day,calendar.monthrange(year,month)[1])
	# calendar.monthrange(year,month) : Returns weekday start of the month (0-6 ~ Mon-Sun) and number of days (28-31) for year, month.

	return datetime.date(year, month, day)
